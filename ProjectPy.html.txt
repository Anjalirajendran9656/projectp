        // Sample questions for suggestions
        const commonQuestions = [
            "How to create a list in Python?",
            "How to write a for loop?",
            "How to define a function?",
            "How to handle exceptions?",
            "How to read a file in Python?",
            "How to install packages with pip?",
            "What is the difference between list and tuple?",
            "How to use dictionaries?",
            "How to import modules?",
            "How to work with strings?",
            "How to create a class?",
            "How to use lambda functions?",
            "How to work with JSON?",
            "How to connect to a database?",
            "How to make HTTP requests?",
            "How to work with dates and time?",
            "How to use regular expressions?",
            "How to create a GUI application?",
            "How to work with CSV files?",
            "How to debug Python code?"
        ];

        const searchInput = document.getElementById('searchInput');
        const suggestionsDiv = document.getElementById('suggestions');

        searchInput.addEventListener('input', function() {
            const query = this.value.toLowerCase().trim();
            
            if (query.length < 2) {
                suggestionsDiv.style.display = 'none';
                return;
            }

            const matches = commonQuestions.filter(question => 
                question.toLowerCase().includes(query)
            );

            if (matches.length > 0) {
                suggestionsDiv.innerHTML = matches
                    .slice(0, 5) // Show max 5 suggestions
                    .map(question => `<div class="suggestion-item" onclick="selectSuggestion('${question}')">${question}</div>`)
                    .join('');
                suggestionsDiv.style.display = 'block';
            } else {
                suggestionsDiv.style.display = 'none';
            }
        });

        // Hide suggestions when clicking outside
        document.addEventListener('click', function(e) {
            if (!searchInput.contains(e.target) && !suggestionsDiv.contains(e.target)) {
                suggestionsDiv.style.display = 'none';
            }
        });

        function selectSuggestion(question) {
            searchInput.value = question;
            suggestionsDiv.style.display = 'none';
            
            // You can add logic here to show relevant FAQ items
            highlightRelevantFAQ(question);
        }

        function highlightRelevantFAQ(question) {
            // Simple keyword matching to highlight relevant FAQ sections
            const keywords = question.toLowerCase();
            const faqItems = document.querySelectorAll('.faq-item');
            
            faqItems.forEach(item => {
                const questionText = item.querySelector('.faq-question').textContent.toLowerCase();
                if (keywords.includes('list') && questionText.includes('list') ||
                    keywords.includes('loop') && questionText.includes('loop') ||
                    keywords.includes('function') && questionText.includes('function') ||
                    keywords.includes('error') && questionText.includes('error')) {
                    
                    item.style.border = '2px solid #ffeb3b';
                    item.style.background = 'rgba(255, 235, 59, 0.1)';
                    
                    // Reset highlight after 3 seconds
                    setTimeout(() => {
                        item.style.border = '1px solid rgba(255, 255, 255, 0.1)';
                        item.style.background = 'rgba(255, 255, 255, 0.08)';
                    }, 3000);
                }
            });
        }

        function toggleAnswer(element) {
            const answer = element.nextElementSibling;
            const icon = element.querySelector('.expand-icon');
            
            if (answer.classList.contains('active')) {
                answer.classList.remove('active');
                icon.classList.remove('rotated');
            } else {
                // Close all other answers
                document.querySelectorAll('.faq-answer.active').forEach(activeAnswer => {
                    activeAnswer.classList.remove('active');
                    activeAnswer.previousElementSibling.querySelector('.expand-icon').classList.remove('rotated');
                });
                
                // Open clicked answer
                answer.classList.add('active');
                icon.classList.add('rotated');
            }
        }

        // Add some interactive features
        document.addEventListener('DOMContentLoaded', function() {
            // Add welcome message
            setTimeout(() => {
                if (searchInput.value === '') {
                    searchInput.placeholder = "Welcome! Try typing 'how to create a list' or 'for loop'...";
                }
            }, 3000);
        });
    </script>
